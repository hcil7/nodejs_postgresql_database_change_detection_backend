CREATE OR REPLACE FUNCTION public.busdata_notify_trigger()
    RETURNS trigger
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE NOT LEAKPROOF
AS $BODY$
 
BEGIN
PERFORM pg_notify (CAST ('update_notification' AS text), row_to_json(NEW)::text);
RETURN NEW;

    END;
    
$BODY$;

ALTER FUNCTION public.busdata_notify_trigger()
    OWNER TO postgres;


//create trigger
CREATE TRIGGER test_table_update_trigger
    AFTER UPDATE 
    ON public.busdata
    FOR EACH ROW
    EXECUTE FUNCTION public.busdata_notify_trigger();
